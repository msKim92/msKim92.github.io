public static void main(String arg[]) {
    
        String [] abc = { "A", "B", "C" ,"C", "C"}; 
        HashMap<String, Integer> hm = new HashMap<>(); 
        
        for(String key : abc) {
        	hm.put(key, hm.getOrDefault(key, 0) + 1);
        }
        System.out.println(hm); 
        //{A=1, B=1, C=3} 
     }
